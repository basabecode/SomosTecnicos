      {/* Modal de Selección de Registro */}
      {isRegisterModalOpen && (
        <div
          className="fixed inset-0 bg-black/50 backdrop-blur-sm z-[100] flex items-center justify-center p-4 animate-in fade-in duration-200"
          onClick={() => setIsRegisterModalOpen(false)}
        >
          <div
            className="bg-white rounded-2xl shadow-2xl max-w-lg w-full overflow-hidden animate-in zoom-in-95 duration-300"
            onClick={(e) => e.stopPropagation()}
          >
            {/* Header del Modal */}
            <div className="bg-gradient-to-r from-[#A50034] to-[#8B1538] px-6 py-5 text-center">
              <h2 className="text-2xl font-bold text-white mb-1">Únete a SomosTécnicos</h2>
              <p className="text-white/90 text-sm">Selecciona cómo deseas registrarte</p>
            </div>

            {/* Opciones de Registro */}
            <div className="p-6 space-y-3">
              {/* Opción Cliente */}
              <Link
                href="/register/customer"
                className="group flex items-center gap-4 bg-white hover:bg-gray-50 rounded-xl p-4 border-2 border-gray-200 hover:border-[#A50034] transition-all duration-300"
                onClick={() => setIsRegisterModalOpen(false)}
              >
                <div className="p-3 bg-[#A50034]/10 rounded-xl group-hover:bg-[#A50034]/20 transition-colors">
                  <Users className="w-6 h-6 text-[#A50034]" />
                </div>

                <div className="flex-1">
                  <h3 className="text-lg font-bold text-[#2C3E50] mb-0.5">Soy Cliente</h3>
                  <p className="text-gray-600 text-xs">
                    Solicita servicios técnicos profesionales
                  </p>
                </div>

                <ArrowRight className="w-5 h-5 text-[#A50034] opacity-0 group-hover:opacity-100 transition-opacity" />
              </Link>

              {/* Opción Técnico */}
              <Link
                href="/register/technician"
                className="group flex items-center gap-4 bg-white hover:bg-gray-50 rounded-xl p-4 border-2 border-gray-200 hover:border-[#2C3E50] transition-all duration-300"
                onClick={() => setIsRegisterModalOpen(false)}
              >
                <div className="p-3 bg-[#2C3E50]/10 rounded-xl group-hover:bg-[#2C3E50]/20 transition-colors">
                  <Wrench className="w-6 h-6 text-[#2C3E50]" />
                </div>

                <div className="flex-1">
                  <h3 className="text-lg font-bold text-[#2C3E50] mb-0.5">Soy Técnico</h3>
                  <p className="text-gray-600 text-xs">
                    Únete a nuestro equipo de profesionales
                  </p>
                </div>

                <ArrowRight className="w-5 h-5 text-[#2C3E50] opacity-0 group-hover:opacity-100 transition-opacity" />
              </Link>
            </div>

            {/* Footer del Modal */}
            <div className="px-6 pb-5 text-center border-t border-gray-100 pt-4">
              <button
                onClick={() => setIsRegisterModalOpen(false)}
                className="text-gray-500 hover:text-gray-700 text-sm font-medium transition-colors"
              >
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}
